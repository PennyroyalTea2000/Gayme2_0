		system("cls");

	}
	//void variable(int h, int a, mob m) {
		//m.health = h;
		//m.attack = a;
	//}

private:
	void fight(mob a) {
		while (a.health > 0) {
			if (choose(a)) 
			a.health--;
			
			//cout << a.name << "получил по ебалу" << endl;
		}
	}

	
	bool choose(mob a) {
		cout << "1.Деремся" << endl << "2. Eбегаем" << endl;
	
		cin >> ch;
		system("cls");
		if (ch == 1) {
			mech();
			cout << a.name << "получил по ебалу" << endl;
			return true;
		}
		else {
			cout << "Убегаешь сверкая пятками" << endl;
			return false;
		}
	}
};

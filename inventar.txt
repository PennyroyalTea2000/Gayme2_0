class weapon {
public:
		string namew;
	int att;
};
class armor {
public:
		string namea;
	int hp;
};
class hero {
public:
		int MAXHP;
	int basHP, basATT;
	void wearW(weapon a) {
		basATT += a.att;
	}
	void wearA(armor a) {
		basHP += a.hp;
		MAXHP = basHP;
	}
	void unwearW(weapon a) {
		basATT -= a.att;
	}
	void unwearA(armor a) {
		basHP -= a.hp;
	}
};
